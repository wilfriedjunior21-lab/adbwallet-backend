import React, { useState, useEffect } from "react";
import { StyleSheet, View, Platform, StatusBar, Text } from "react-native";
import { WebView } from "react-native-webview";
import NetInfo from "@react-native-community/netinfo";

export default function App() {
  const [isConnected, setIsConnected] = useState(true);

  // Vérification de la connexion internet en temps réel
  useEffect(() => {
    const unsubscribe = NetInfo.addEventListener(state => {
      setIsConnected(state.isConnected);
    });
    return () => unsubscribe();
  }, []);

  // Écran de secours si l'utilisateur n'a pas internet
  if (!isConnected) {
    return (
      <View style={styles.errorContainer}>
        <StatusBar barStyle="light-content" backgroundColor="#000" />
        <Text style={styles.errorEmoji}>📡</Text>
        <Text style={styles.errorText}>Connexion interrompue</Text>
        <Text style={styles.errorSubText}>Veuillez vérifier votre accès réseau pour utiliser ADBWALLET.</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <StatusBar barStyle="light-content" backgroundColor="#000" />

      <WebView
        source={{ uri: "https://adbwallet-frontend.pages.dev" }}
        style={styles.webview}
        javaScriptEnabled={true}
        domStorageEnabled={true}
        startInLoadingState={true}
        // Couleur de fond pendant le chargement pour éviter le flash blanc
        backgroundColor="#000" 
      />

      <View style={styles.footer}>
        <Text style={styles.footerText}>© COPYRIGHT ADBWALLET 2025</Text>
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: "#000",
    paddingTop: Platform.OS === "android" ? StatusBar.currentHeight : 0,
  },
  webview: {
    flex: 1,
  },
  footer: {
    backgroundColor: "#000",
    paddingVertical: 12,
    borderTopWidth: 0.5,
    borderTopColor: "#1e293b",
    alignItems: "center",
  },
  footerText: {
    color: "#64748b",
    fontSize: 10,
    fontWeight: "600",
    letterSpacing: 1,
  },
  // Styles pour l'écran Hors-ligne
  errorContainer: {
    flex: 1,
    backgroundColor: "#000",
    justifyContent: "center",
    alignItems: "center",
    padding: 20,
  },
  errorEmoji: {
    fontSize: 50,
    marginBottom: 20,
  },
  errorText: {
    color: "#fff",
    fontSize: 18,
    fontWeight: "bold",
    marginBottom: 10,
  },
  errorSubText: {
    color: "#64748b",
    textAlign: "center",
    lineHeight: 20,
  },
});
